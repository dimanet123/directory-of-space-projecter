Ручное вычисление и все выкладки представлены в [[Определение потребной скорости для схода КА]]

# Определение потребной скорости для схода с круговой орбиты

import numpy as np

# Объявление стартовых параметров
R_planet = 6371
mu = 398600
h_atm = 100
R_atm = R_planet + h_atm


def needed_V(h_start, h_end):

    # Получение радиусов стартовых орбит
    R_start = R_planet + h_start
    R_end = R_planet + h_end

    # Получение большей полуоси новой орбиты
    a = (R_start + R_end)/2

    # Вычисление начальной скорости и новой скорости в апогее
    V0 = np.sqrt(mu/R_start)
    V1 = np.sqrt(mu*(2/R_start - 1/a))

    # Изменение скорости
    deltaV = V1-V0

    # Скорость входа в атомсферу
    V_in = np.sqrt(mu*(2/R_atm - 1/a))

    # Угол входа в атмосферу
    theta = np.arccos((R_start*V1)/(R_atm*V_in))
    theta = np.rad2deg(theta)

    # Вывод параметров
    print(f'Тормозной имульс для схода с орбиты = {deltaV}')
    print(f'Скорость входа в атосферу = {V_in}')
    print(f'Угол входа в атмосферу = {theta}')


# Вызов функции для начальной высоты 500км и конечной 80 км
needed_V(500, 80)
